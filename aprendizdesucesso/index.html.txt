<script>
let notaInovacao = null;
let notaCriatividade = null;

document.querySelectorAll('#inovacao .nota-btn').forEach(btn => {
  btn.addEventListener('click', () => {
    document.querySelectorAll('#inovacao .nota-btn').forEach(b => b.classList.remove('selected'));
    btn.classList.add('selected');
    notaInovacao = btn.textContent;
  });
});

document.querySelectorAll('#criatividade .nota-btn').forEach(btn => {
  btn.addEventListener('click', () => {
    document.querySelectorAll('#criatividade .nota-btn').forEach(b => b.classList.remove('selected'));
    btn.classList.add('selected');
    notaCriatividade = btn.textContent;
  });
});

function salvarAvaliacao() {
  const nome = document.getElementById('empresa').value;
  const resultados = document.getElementById('resultados');

  if (!nome || notaInovacao === null || notaCriatividade === null) {
    alert('Preencha todos os campos e notas!');
    return;
  }

  const empresaDiv = document.createElement('div');
  empresaDiv.classList.add('empresa-item');
  empresaDiv.innerHTML = `
    <h3>${nome}</h3>
    <p class="nota-grande">Inovação: ${notaInovacao}</p>
    <p class="nota-grande">Criatividade: ${notaCriatividade}</p>
  `;
  resultados.appendChild(empresaDiv);

  // Limpar para próxima empresa
  document.getElementById('empresa').value = '';
  document.querySelectorAll('.nota-btn').forEach(b => b.classList.remove('selected'));
  notaInovacao = null;
  notaCriatividade = null;
}
</script>
